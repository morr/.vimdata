  def filter_ns(ns, attrs)
    return attrs if attrs.nil? or attrs.empty?
    newattrs = {}
    attrs.each do |key, value|
      if (NSParseRegexp =~ key)
        # '' means 'default namespace'.
        tag = $1 || ''
        ns.assign(value, tag)
      else
        newattrs[key] = value
      end
    end
    newattrs
  end
