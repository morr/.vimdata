  def test_openSchema
    con = nil
    begin
      con = WIN32OLE.new('ADODB.Connection')
      con.connectionString = "Provider=MSDASQL;Extended Properties="
      con.connectionString +="\"DRIVER={Microsoft Text Driver (*.txt; *.csv)};DBQ=#{@path}\""
      con.open
    rescue
      con = nil
    end
    if con
      rs = con.openSchema(4, [nil,nil,"DUMMY", "TABLE"])
      assert(rs)
    end
  end
