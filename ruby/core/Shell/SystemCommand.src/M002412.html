    def notify(*opts, &block)
      Thread.exclusive do
        @shell.notify(*opts) {|mes|
          yield mes if iterator?

          mes.gsub!("%id", "#{@command}:##{@pid}")
          mes.gsub!("%name", "#{@command}")
          mes.gsub!("%pid", "#{@pid}")
        }
      end
    end
