  def display_list(b, e, file, line)
    stdout.printf "[%d, %d] in %s\n", b, e, file
    if lines = SCRIPT_LINES__[file] and lines != true
      n = 0
      b.upto(e) do |n|
        if n > 0 && lines[n-1]
          if n == line
            stdout.printf "=> %d  %s\n", n, lines[n-1].chomp
          else
            stdout.printf "   %d  %s\n", n, lines[n-1].chomp
          end
        end
      end
    else
      stdout.printf "No sourcefile available for %s\n", file
    end
  end
