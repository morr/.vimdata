  def collect_complextypes
    result = @anontypes.dup
    if @types
      @types.schemas.each do |schema|
        result.concat(schema.collect_complextypes)
      end
    end
    @imports.each do |import|
      result.concat(import.content.collect_complextypes)
    end
    result
  end
