  def collect_elements
    result = XSD::NamedElements.new
    if @types
      @types.schemas.each do |schema|
        result.concat(schema.collect_elements)
      end
    end
    @imports.each do |import|
      result.concat(import.content.collect_elements)
    end
    result
  end
