    def self.select_message(receiver, message, candidates)
      candidates.grep(/^#{message}/).collect do |e|
        case e
        when /^[a-zA-Z_]/
          receiver + "." + e
        when /^[0-9]/
        when *Operators
          #receiver + " " + e
        end
      end
    end
