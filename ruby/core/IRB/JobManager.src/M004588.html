    def inspect
      ary = []
      @jobs.each_index do
        |i|
        th, irb = @jobs[i]
        next if th.nil?

        if th.alive?
          if th.stop?
            t_status = "stop"
          else
            t_status = "running"
          end
        else
          t_status = "exited"
        end
        ary.push format("#%d->%s on %s (%s: %s)",
                        i, 
                        irb.context.irb_name, 
                        irb.context.main,
                        th,
                        t_status)
      end
      ary.join("\n")
    end
