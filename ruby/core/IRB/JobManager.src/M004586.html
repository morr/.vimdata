    def search(key)
      case key
      when Integer
        @jobs[key]
      when Irb
        @jobs.find{|k, v| v.equal?(key)}
      when Thread
        @jobs.assoc(key)
      else
        assoc = @jobs.find{|k, v| v.context.main.equal?(key)}
        IRB.fail NoSuchJob, key if assoc.nil?
        assoc
      end
    end
