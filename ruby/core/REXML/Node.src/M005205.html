                def to_s indent=nil
      unless indent.nil?
        Kernel.warn( "#{self.class.name}.to_s(indent) parameter is deprecated" )
        f = REXML::Formatters::Pretty.new( indent )
        f.write( self, rv, indent )
      else
        f = REXML::Formatters::Default.new
        f.write( self, rv = "" )
      end
      return rv
                end
