                        def initialize src
                                super()
                                if src.match( PUBLIC )
                                        md = src.match( PUBLIC, true )
                                elsif src.match( SYSTEM )
                                        md = src.match( SYSTEM, true )
                                else
                                        raise ParseException.new( "error parsing notation: no matching pattern", src )
                                end
                                @name = md[1]
                                @middle = md[2]
                                @rest = md[3]
                        end
