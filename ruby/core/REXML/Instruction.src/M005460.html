                def initialize(target, content=nil)
                        if target.kind_of? String
                                super()
                                @target = target
                                @content = content
                        elsif target.kind_of? Instruction
                                super(content)
                                @target = target.target
                                @content = target.content
                        end
                        @content.strip! if @content
                end
