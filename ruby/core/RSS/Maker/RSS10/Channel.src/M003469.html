        def to_rss(rss)
          set = false
          if @about
            channel = RDF::Channel.new(@about)
            set = setup_values(channel)
            if set
              channel.dc_dates.clear
              rss.channel = channel
              setup_items(rss)
              setup_image(rss)
              setup_textinput(rss)
              setup_other_elements(rss)
            end
          end

          if (!@about or !set) and variable_is_set?
            raise NotSetError.new("maker.channel", not_set_required_variables)
          end
        end
