          def to_rss(rss, channel)
            unless @hours.empty?
              skipHours = Rss::Channel::SkipHours.new
              channel.skipHours = skipHours
              @hours.each do |hour|
                hour.to_rss(rss, skipHours.hours)
              end
            end
          end
