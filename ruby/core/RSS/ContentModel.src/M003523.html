    def self.append_features(klass)
      super

      klass.install_must_call_validator(CONTENT_PREFIX, CONTENT_URI)
      %w(encoded).each do |name|
        klass.install_text_element(name, CONTENT_URI, "?",
                                   "#{CONTENT_PREFIX}_#{name}")
      end
    end
