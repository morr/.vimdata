  def ungetc(c = nil)
    if @here_readed.empty?
      c2 = @readed.pop
    else
      c2 = @here_readed.pop
    end
    c = c2 unless c
    @rests.unshift c #c = 
      @seek -= 1
    if c == "\n"
      @line_no -= 1 
      if idx = @readed.reverse.index("\n")
        @char_no = @readed.size - idx
      else
        @char_no = @base_char_no + @readed.size
      end
    else
      @char_no -= 1
    end
  end
