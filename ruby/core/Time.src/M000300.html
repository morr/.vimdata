/*
 * call-seq:
 *   Time._load(string)   => time
 *
 * Unmarshal a dumped +Time+ object.
 */

static VALUE
time_load(klass, str)
    VALUE klass, str;
{
    VALUE time = time_s_alloc(klass);

    if (FL_TEST(str, FL_EXIVAR)) {
        rb_copy_generic_ivar(time, str);
        FL_SET(time, FL_EXIVAR);
    }
    time_mload(time, str);
    return time;
}
