    def get(remotefile, localfile = File.basename(remotefile),
            blocksize = DEFAULT_BLOCKSIZE, &block) # :yield: data
      unless @binary
        gettextfile(remotefile, localfile, &block)
      else
        getbinaryfile(remotefile, localfile, blocksize, &block)
      end
    end
