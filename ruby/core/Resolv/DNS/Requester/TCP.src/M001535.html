        def initialize(host, port=Port)
          super()
          @host = host
          @port = port
          @sock = TCPSocket.new(@host, @port)
          @sock.fcntl(Fcntl::F_SETFD, Fcntl::FD_CLOEXEC) if defined? Fcntl::F_SETFD
          @senders = {}
        end
