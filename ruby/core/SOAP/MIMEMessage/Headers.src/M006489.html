    def parse_line(line)
      if /^\A([^\: \t]+):\s*(.+)\z/ =~ line
        header = parse_rhs($2.strip)
        header.key = $1.strip
        self[header.key.downcase] = header
      else
        raise RuntimeError.new("unexpected header line: #{line.inspect}")
      end
    end
