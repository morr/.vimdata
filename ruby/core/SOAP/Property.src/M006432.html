  def local_assign(key, value)
    check_lock(key)
    if @locked
      if propkey?(value)
        raise FrozenError.new("cannot add any key to locked property")
      elsif propkey?(@store[key])
        raise FrozenError.new("cannot override any key in locked property")
      end
    end
    @store[key] = value
  end
