      def shape_bounds(kind=nil)
        if kind
          ret = tk_call('shape', 'bounds', @path, "-#{kind}")
        else
          ret = tk_call('shape', 'bounds', @path)
        end
        if ret == ""
          nil
        else
          list(ret)
        end
      end

          tk_call('shape', 'get', @path, "-#{kind}", x, y)

          tk_call('shape', 'get', @path, x, y)
        end

      end
