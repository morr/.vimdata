        def parse(contents)
          contents.each{|elem|
            name,ty,num,enc,dec = parse_elem(elem)
            @names.push(name)
            @ty[name]  = ty
            @len[name] = num
            @enc[name] = enc
            @dec[name] = dec
            if( num )
              @tys += "#{ty}#{num}"
            else
              @tys += ty
            end
          }
          @size = DL.sizeof(@tys)
        end

        def parse_elem(elem)
          elem.strip!
          case elem
          when /^([\w\d_\*]+)([\*\s]+)([\w\d_]+)$/
            ty = ($1 + $2).strip
            name = $3
            num = nil;
          when /^([\w\d_\*]+)([\*\s]+)([\w\d_]+)\[(\d+)\]$/
            ty = ($1 + $2).strip
            name = $3
            num = $4.to_i
          else
            raise(RuntimeError, "invalid element: #{elem}")
          end
          ty,enc,dec = @types.encode_struct_type(ty)
          if( !ty )
            raise(TypeError, "unsupported type: #{ty}")
          end
          return [name,ty,num,enc,dec]
        end
