  def dump(types)
    @types = types
    map_cache = []
    map = ""
    @types.each do |type|
      if map_cache.index(type).nil?
        map_cache << type
        if type.namespace != XSD::Namespace
          if typemap = dump_typemap(type)
            map << typemap
          end
        end
      end
   end
    return map
  end
