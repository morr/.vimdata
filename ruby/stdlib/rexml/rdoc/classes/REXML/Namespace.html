                def has_name?( other, ns=nil )
                        if ns
                                return (namespace() == ns and name() == other)
                        elsif other.include? ":"
                                return fully_expanded_name == other
                        else
                                return name == other
                        end
                end

                def name=( name )
                        @expanded_name = name
                        name =~ NAMESPLIT
                        if $1
                                @prefix = $1
                        else
                                @prefix = ""
                                @namespace = ""
                        end
                        @name = $2
                end
