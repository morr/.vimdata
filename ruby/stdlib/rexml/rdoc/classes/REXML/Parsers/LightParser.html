                        def parse
                                root = context = [ :document ]
                                while true
                                        event = @parser.pull
                                        case event[0]
                                        when :end_document
                                                break
                                        when :end_doctype
                                                context = context[1]
                                        when :start_element, :start_doctype
                                                new_node = event
                                                context << new_node
                                                new_node[1,0] = [context]
                                                context = new_node
                                        when :end_element, :end_doctype
                                                context = context[1]
                                        else
                                                new_node = event
                                                context << new_node
                                                new_node[1,0] = [context]
                                        end
                                end
                                root
                        end
